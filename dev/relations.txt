Create table LOCATION(
  LOCA_ID INTEGER PRIMARY KEY AUTO_INCREMENT, 
  LOCA_NAME VARCHAR(12) NOT NULL
);

CREATE TABLE COMPANY(
  CO_ID VARCHAR(2) PRIMARY KEY,
  LOCA_ID INTEGER NOT NULL,
  CO_NAME VARCHAR(20) NOT NULL,
  FOREIGN KEY (LOCA_ID) REFERENCES LOCATION(LOCA_ID)
  CHECK (CO_ID in ('A', 'B', 'C', 'D', 'E', 'F', 'G', 'H', 'I', 'J', 'K', 'L', 'M', 'N', 'O', 'P', 'BD', 'R', 'S', 'T', 'PB')
);

CREATE TABLE CADET(
  CWID INTEGER PRIMARY KEY,
  CO_ID VARCHAR(2) NOT NULL,
  F_NAME VARCHAR(15) NOT NULL,
  M_INITIAL VARCHAR(1),
  L_NAME VARCHAR(15) NOT NULL,
  ROOM_NUM INTEGER NOT NULL,
  PHONE_NUM VARCHAR(13) NOT NULL UNIQUE,
  EMAIL VARCHAR(30) NOT NULL UNIQUE,
  RANK_ID VARCHAR(5) NOT NULL,
  FOREIGN KEY (CO_ID) REFERENCES COMPANY(CO_ID),
  FOREIGN KEY (RANK_ID) REFERENCES RANK(RANK_ID)
  CHECK (CWID BETWEEN 10000000 AND 99999999)
);

CREATE TABLE RANK(
  RANK_ID INTEGER PRIMARY KEY AUTO_INCREMENT,
  RANK_NAME VARCHAR(6) NOT NULL,
  BOL_SWORD CHAR NOT NULL
);

CREATE TABLE ENROLLMENT(
  ENROLL_ID INTEGER PRIMARY KEY AUTO_INCREMENT,
  CWID INTEGER NOT NULL,
  CLASS_ID INTEGER NOT NULL,
  FOREIGN KEY (CWID) REFERENCES CADET(CWID),
  FOREIGN KEY (CLASS_ID) REFERENCES CLASS(CLASS_ID)
);

CREATE TABLE GUARD_ASSIGNMENT(
  BAT_ASGN_ID INTEGER PRIMARY KEY AUTO_INCREMENT,
  CWID INTEGER NOT NULL,
  GUARD_DATE DATE NOT NULL,
  LOCA_ID INTEGER NOT NULL,
  GUARD_TIME INTEGER NOT NULL,
  FOREIGN KEY (CWID) REFERENCES CADET(CWID),
  FOREIGN KEY (LOCA_ID) REFERENCES LOCATION(LOCA_ID)
);

CREATE TABLE CLASS(
  CLASS_ID INTEGER PRIMARY KEY AUTO_INCREMENT,
  CRN INTEGER NOT NULL,
  DAY CHAR,
  START_TIME INTEGER,
  END_TIME INTEGER,
  CHECK (DAY in ('M', 'T', 'W', 'R', 'F')),
  CHECK (CRN BETWEEN 10000, 99999)
);
