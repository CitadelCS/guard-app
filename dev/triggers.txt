CREATE TRIGGER TRG_UPDATE_ADDCOUNT
AFTER INSERT ON GUARD_ASSIGNMENT
FOR EACH ROW
    UPDATE CADET
    SET CADET.COUNT = CADET.COUNT + 1
    WHERE GUARD_ASSIGNMENT.CWID = CADET.CWID;

CREATE TRIGGER TRG_UPDATE_SUBCOUNT
AFTER DELETE ON GUARD_ASSIGNMENT
FOR EACH ROW
    UPDATE CADET
    SET CADET.COUNT = CADET.COUNT - 1
    WHERE GUARD_ASSIGNMENT.CWID = CADET.CWID;
